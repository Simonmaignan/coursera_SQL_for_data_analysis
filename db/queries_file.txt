-- Find states where both honey and milk were produced in 2022. Did State_ANSI "35" produce both honey and milk in 2022?
SELECT
    m.State_ANSI
FROM
    honey_production AS h
    INNER JOIN milk_production AS m ON h.State_ANSI = m.State_ANSI
WHERE
    m.Year = 2022
    AND m.State_ANSI = 35
