-- Find the total yogurt production for states that also produced cheese in 2022.
-- Answer = 1171095000
SELECT
    sum(Value)
FROM
    yogurt_production
WHERE
    Year = 2022
    AND State_ANSI IN (
        SELECT
            State_ANSI
        FROM
            cheese_production
        WHERE
            Year = 2022
    )
