-- Find the total yogurt production for states that also produced cheese in 2022.
SELECT
    sum(y.Value)
    -- c.Value,
    -- y.Value,
    -- c.State_ANSI,
    -- y.Year
FROM
    cheese_production AS c
    INNER JOIN yogurt_production AS y ON c.State_ANSI = y.State_ANSI
WHERE
    y.Year = 2022
