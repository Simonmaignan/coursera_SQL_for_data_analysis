-- Find states where both honey and milk were produced in 2022.
-- Did State_ANSI "35" produce both honey and milk in 2022?
SELECT
    m.State_ANSI,
    h.State_ANSI,
    m.Year,
    h.Year
FROM
    milk_production as m
    INNER JOIN honey_production as h ON m.Year = h.Year
WHERE
    m.State_ANSI = 35
    and h.State_ANSI = 35
    -- and h.Year = 2022
